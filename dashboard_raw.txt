<script>
  const es = new EventSource("/events");
  const liveCounts = {};

  es.onmessage = e => {
    const [camId, count, time, name] = e.data.split("|");
    const n = parseInt(count);
    liveCounts[camId] = n;

    const badge = document.getElementById(`count-${camId}`);
    if (badge) {
      badge.textContent = `${count} person${n !== 1 ? "s" : ""}`;
      badge.className = "cam-count" + (n > 0 ? " active" : "");
    }

    const total = Object.values(liveCounts).reduce((a, b) => a + b, 0);
    document.getElementById("total-live").textContent = total;

    // only log when persons are detected
    if (n > 0) {
      const li = document.createElement("li");
      li.innerHTML = `<span class="log-cam">${name}</span>
                      <span class="log-count">${count}p</span>
                      <span class="log-time">${time}</span>`;
      const list = document.getElementById("log-list");
      list.prepend(li);
      if (list.children.length > 30) list.lastChild.remove();
    }
  };
</script>
